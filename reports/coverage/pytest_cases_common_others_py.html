94def assert_exception(expected # type: ExpectedError 

95 ): 

96 """ 

97 A context manager to check that some bit of code raises an exception. Sometimes it might be more 

98 handy than `with pytest.raises():`. 

99 

100 `expected` can be: 

101 

102 - an expected error type, in which case `isinstance(caught, expected)` will be used for validity checking 

103 

104 - an expected error representation pattern (a regex pattern string), in which case 

105 `expected.match(repr(caught))` will be used for validity checking 

106 

107 - an expected error instance, in which case BOTH `isinstance(caught, type(expected))` AND 

108 `caught == expected` will be used for validity checking 

109 

110 - an error validation callable, in which case `expected(caught) is not False` will be used for validity 

111 checking 

112 

113 Upon failure, this raises an `ExceptionCheckingError` (a subclass of `AssertionError`) 

114 

115 ```python 

116 # good type - ok 

117 with assert_exception(ValueError): 

118 raise ValueError() 

119 

120 # good type - inherited - ok 

121 class MyErr(ValueError): 

122 pass 

123 with assert_exception(ValueError): 

124 raise MyErr() 

125 

126 # no exception - raises ExceptionCheckingError 

127 with assert_exception(ValueError): 

128 pass 

129 

130 # wrong type - raises ExceptionCheckingError 

131 with assert_exception(ValueError): 

132 raise TypeError() 

133 

134 # good repr pattern - ok 

135 with assert_exception(r"ValueError\('hello'[,]+\)"): 

136 raise ValueError("hello") 

137 

138 # good instance equality check - ok 

139 class MyExc(Exception): 

140 def __eq__(self, other): 

141 return vars(self) == vars(other) 

142 with assert_exception(MyExc('hello')): 

143 raise MyExc("hello") 

144 

145 # good equality but wrong type - raises ExceptionCheckingError 

146 with assert_exception(MyExc('hello')): 

147 raise Exception("hello") 

148 ``` 

149 

150 :param expected: an exception type, instance, repr string pattern, or a callable 

151 """ 

152 return AssertException(expected)
