45 def _fixture_product(fixtures_dest,
46 name, # type: str
47 fixtures_or_values,
48 fixture_positions,
49 scope="function", # type: str
50 unpack_into=None, # type: Iterable[str]
51 autouse=False, # type: bool
52 hook=None, # type: Callable[[Callable], Callable]
53 caller=None, # type: Callable
54 **kwargs):
55 """
56 Internal implementation for fixture products created by pytest parametrize plus.
57  
58 :param fixtures_dest:
59 :param name:
60 :param fixtures_or_values:
61 :param fixture_positions:
62 :param idstyle:
63 :param scope:
64 :param ids:
65 :param unpack_into:
66 :param autouse:
67 :param kwargs:
68 :return:
69 """
70 # test the `fixtures` argument to avoid common mistakes
71 if not isinstance(fixtures_or_values, (tuple, set, list)):
72 raise TypeError("fixture_product: the `fixtures_or_values` argument should be a tuple, set or list")
73 else:
74 has_lazy_vals = any(is_lazy_value(v) for v in fixtures_or_values)
75  
76 _tuple_size = len(fixtures_or_values)
77  
78 # first get all required fixture names
79 f_names = [None] * _tuple_size
80 for f_pos in fixture_positions:
81 # possibly get the fixture name if the fixture symbol was provided
82 f = fixtures_or_values[f_pos]
83 if isinstance(f, fixture_ref):
84 f = f.fixture
85 # and remember the position in the tuple
86 f_names[f_pos] = get_fixture_name(f)
87  
88 # remove duplicates by making it an ordered set
89 all_names = remove_duplicates((n for n in f_names if n is not None))
90 if len(all_names) < 1:
91 raise ValueError("Empty fixture products are not permitted")
92  
93 def _tuple_generator(request, all_fixtures):
94 for i in range(_tuple_size):
95 fix_at_pos_i = f_names[i]
96 if fix_at_pos_i is None:
97 # fixed value
98 # note: wouldn't it be almost as efficient but more readable to *always* call handle_lazy_args?
99 yield get_lazy_args(fixtures_or_values[i], request) if has_lazy_vals else fixtures_or_values[i]
100 else:
101 # fixture value
102 yield all_fixtures[fix_at_pos_i]
103  
104 # then generate the body of our product fixture. It will require all of its dependent fixtures
105 @with_signature("(request, %s)" % ', '.join(all_names))
106 def _new_fixture(request, **all_fixtures):
107 return tuple(_tuple_generator(request, all_fixtures))
108  
109 _new_fixture.__name__ = name
110  
111 # finally create the fixture per se.
112 # WARNING we do not use pytest.fixture but fixture so that NOT_USED is discarded
113 f_decorator = fixture(scope=scope, autouse=autouse, hook=hook, **kwargs)
114 fix = f_decorator(_new_fixture)
115  
116 # Dynamically add fixture to caller's module as explained in https://github.com/pytest-dev/pytest/issues/2424
117 check_name_available(fixtures_dest, name, if_name_exists=WARN, caller=caller)
118 setattr(fixtures_dest, name, fix)
119  
120 # if unpacking is requested, do it here
121 if unpack_into is not None:
122 # note that as for fixture unions, we can not expose the `in_cls` parameter.
123 # but there is an easy workaround if unpacking is needed: call unpack_fixture separately
124 _make_unpack_fixture(fixtures_dest, argnames=unpack_into, fixture=name, hook=hook, in_cls=False)
125  
126 return fix

133 class fixture_ref(object): # noqa
134 """
135 A reference to a fixture, to be used in `@parametrize`.
136 You can create it from a fixture name or a fixture object (function).
137 """
138 __slots__ = 'fixture', 'theoretical_size', '_id'
139  
140 def __init__(self,
141 fixture, # type: Union[str, Callable]
142 id=None, # type: str # noqa
143 ):
144 """
145  
146 :param fixture: the name of the fixture to reference, or the fixture function itself
147 :param id: an optional custom id to override the fixture name in ids.
148 """
149 self.fixture = get_fixture_name(fixture)
150 self._id = id
151 self.theoretical_size = None # we dont know yet, will be filled by @parametrize
152  
153 def get_name_for_id(self):
154 """return the name to use in ids."""
155 return self._id if self._id is not None else self.fixture
156  
157 def __str__(self):
158 # used in mini_idval for example
159 return self.get_name_for_id()
160  
161 def __repr__(self):
162 if self._id is not None:
163 return "fixture_ref<%s, id=%s>" % (self.fixture, self._id)
164 else:
165 return "fixture_ref<%s>" % self.fixture
166  
167 def _check_iterable(self):
168 """Raise a TypeError if this fixture reference is not iterable, that is, it does not represent a tuple"""
169 if self.theoretical_size is None:
170 raise TypeError("This `fixture_ref` has not yet been initialized, so it cannot be unpacked/iterated upon. "
171 "This is not supposed to happen when a `fixture_ref` is used correctly, i.e. as an item in"
172 " the `argvalues` of a `@parametrize` decorator. Please check the documentation for "
173 "details.")
174 if self.theoretical_size == 1:
175 raise TypeError("This fixture_ref does not represent a tuple of arguments, it is not iterable")
176  
177 def __len__(self):
178 self._check_iterable()
179 return self.theoretical_size
180  
181 def __getitem__(self, item):
182 """
183 Returns an item in the tuple described by this fixture_ref.
184 This is just a facade, a FixtureRefItem.
185 Note: this is only used when a custom `idgen` is passed to @parametrized
186 """
187 self._check_iterable()
188 return FixtureRefItem(self, item)

406 @classmethod
407 def create(cls,
408 new_fixture_host, # type: Union[Type, ModuleType]
409 test_func, # type: Callable
410 param_union_name, # type: str
411 argnames, # type: Sequence[str]
412 from_i, # type: int
413 to_i, # type: int
414 argvalues, # type: Any
415 ids, # type: Union[Sequence[str], Callable]
416 hook=None, # type: Callable
417 debug=False # type: bool
418 ):
419 # type: (...) -> MultiParamAlternative
420 """
421 Creates an alternative for fixture union `param_union_name`, to handle a group of consecutive parameters
422 argvalues[from_i:to_i] in @parametrize. Note that here the received `argvalues` should be already sliced
423  
424 This alternative will refer to a newly created fixture in `new_fixture_host`, that will be parametrized to
425 return each of `argvalues`.
426  
427 :param new_fixture_host:
428 :param test_func:
429 :param param_union_name:
430 :param argnames:
431 :param from_i:
432 :param to_i:
433 :param argvalues:
434 :param hook:
435 :param debug:
436 :return:
437 """
438 nb_params = len(argnames)
439 param_names_str = '_'.join(argnames).replace(' ', '')
440  
441 # Create a unique fixture name
442 p_fix_name = "%s_%s_is_P%stoP%s" % (test_func.__name__, param_names_str, from_i, to_i - 1)
443 p_fix_name = check_name_available(new_fixture_host, p_fix_name, if_name_exists=CHANGE, caller=parametrize)
444  
445 if debug:
  • T001 Print found.
446 print(" - Creating new fixture %r to handle parameters %s to %s" % (p_fix_name, from_i, to_i - 1))
447  
448 # Create the fixture
449 # - it will be parametrized to take all the values in argvalues
450 # - therefore it will use the custom ids and marks if any
451 # - it will be unique (not unfolded) so if there are more than 1 argnames we have to add a layer of tuple in the
452 # values
453  
454 if nb_params > 1:
455 # we have to create a tuple around the vals because we have a SINGLE parameter that is a tuple
456 unmarked_argvalues = []
457 new_argvals = []
458 for v in argvalues:
459 if is_marked_parameter_value(v):
460 # transform the parameterset so that it contains a tuple of length 1
461 vals = get_marked_parameter_values(v, nbargs=nb_params)
462 if nb_params == 1:
463 vals = vals[0]
464 unmarked_argvalues.append(vals)
465 new_argvals.append(ParameterSet((vals,),
466 id=get_marked_parameter_id(v),
467 marks=get_marked_parameter_marks(v)))
468 else:
469 # nothing special to do since there is no pytest.param here
470 new_argvals.append(v)
471 unmarked_argvalues.append(v)
472 argvalues = new_argvals
473  
474 # we also have to generate the ids correctly "as if they were multiple"
475 try:
476 iter(ids)
477 except TypeError:
478 if ids is not None:
479 ids = ["-".join(ids(vi) for vi in v) for v in unmarked_argvalues]
480 else:
481 ids = [mini_idvalset(argnames, vals, i) for i, vals in enumerate(unmarked_argvalues)]
482  
483 _create_param_fixture(new_fixture_host, argname=p_fix_name, argvalues=argvalues, ids=ids, hook=hook,
484 debug=debug)
485  
486 # Create the corresponding alternative
487 # note: as opposed to SingleParamAlternative, no need to move the custom id/marks to the ParamAlternative
488 # since they are set on the created parametrized fixture above
489 return MultiParamAlternative(union_name=param_union_name, alternative_name=p_fix_name, argnames=argnames,
490 param_index_from=from_i, param_index_to=to_i, decorated=test_func)

618 def parametrize(argnames=None, # type: Union[str, Tuple[str], List[str]]
619 argvalues=None, # type: Iterable[Any]
620 indirect=False, # type: bool
621 ids=None, # type: Union[Callable, Iterable[str]]
622 idstyle=None, # type: Union[str, Callable]
623 idgen=_IDGEN, # type: Union[str, Callable]

626 hook=None, # type: Callable[[Callable], Callable]
627 debug=False, # type: bool
628 **args):
629 # type: (...) -> Callable[[T], T]
630 """
631 Equivalent to `@pytest.mark.parametrize` but also supports
632  
633 (1) new alternate style for argnames/argvalues. One can also use `**args` to pass additional `{argnames: argvalues}`
634 in the same parametrization call. This can be handy in combination with `idgen` to master the whole id template
635 associated with several parameters. Note that you can pass coma-separated argnames too, by de-referencing a dict:
636 e.g. `**{'a,b': [(0, True), (1, False)], 'c': [-1, 2]}`.
637  
638 (2) new alternate style for ids. One can use `idgen` instead of `ids`. `idgen` can be a callable receiving all
639 parameters at once (`**args`) and returning an id ; or it can be a string template using the new-style string
640 formatting where the argnames can be used as variables (e.g. `idgen=lambda **args: "a={a}".format(**args)` or
641 `idgen="my_id where a={a}"`). The special `idgen=AUTO` symbol can be used to generate a default string template
642 equivalent to `lambda **args: "-".join("%s=%s" % (n, v) for n, v in args.items())`. This is enabled by default
643 if you use the alternate style for argnames/argvalues (e.g. if `len(args) > 0`), and if there are no `fixture_ref`s
644 in your argvalues.
645  
646 (3) new possibilities in argvalues:
647  
648 - one can include references to fixtures with `fixture_ref(<fixture>)` where <fixture> can be the fixture name or
649 fixture function. When such a fixture reference is detected in the argvalues, a new function-scope "union"
650 fixture will be created with a unique name, and the test function will be wrapped so as to be injected with the
651 correct parameters from this fixture. Special test ids will be created to illustrate the switching between the
652 various normal parameters and fixtures. You can see debug print messages about all fixtures created using
653 `debug=True`
654  
655 - one can include lazy argvalues with `lazy_value(<valuegetter>, [id=..., marks=...])`. A `lazy_value` is the same
656 thing than a function-scoped fixture, except that the value getter function is not a fixture and therefore can
657 neither be parametrized nor depend on fixtures. It should have no mandatory argument.
658  
659 Both `fixture_ref` and `lazy_value` can be used to represent a single argvalue, or a whole tuple of argvalues when
660 there are several argnames. Several of them can be used in a tuple.
661  
662 Finally, `pytest.param` is supported even when there are `fixture_ref` and `lazy_value`.
663  
664 An optional `hook` can be passed, to apply on each fixture function that is created during this call. The hook
665 function will be called every time a fixture is about to be created. It will receive a single argument (the
666 function implementing the fixture) and should return the function to use. For example you can use `saved_fixture`
667 from `pytest-harvest` as a hook in order to save all such created fixtures in the fixture store.
668  
669 :param argnames: same as in pytest.mark.parametrize
670 :param argvalues: same as in pytest.mark.parametrize except that `fixture_ref` and `lazy_value` are supported
671 :param indirect: same as in pytest.mark.parametrize. Note that it is not recommended and is not guaranteed to work
672 in complex parametrization scenarii.
673 :param ids: same as in pytest.mark.parametrize. Note that an alternative way to create ids exists with `idgen`. Only
674 one non-None `ids` or `idgen should be provided.
675 :param idgen: an id formatter. Either a string representing a template, or a callable receiving all argvalues
676 at once (as opposed to the behaviour in pytest ids). This alternative way to generate ids can only be used when
677 `ids` is not provided (None). You can use the special `AUTO` formatter to generate an automatic id with
678 template <name>=<value>-<name2>=<value2>-etc. `AUTO` is enabled by default if you use the alternate style for
679 argnames/argvalues (e.g. if `len(args) > 0`), and if there are no `fixture_ref`s in your argvalues.
680 :param auto_refs: a boolean. If this is `True` (default), argvalues containing fixture symbols will automatically
681 be wrapped into a `fixture_ref`, for convenience.
682 :param idstyle: This is mostly for debug. Style of ids to be used in the "union" fixtures generated by
683 `@parametrize` if at least one `fixture_ref` is found in the argvalues. `idstyle` possible values are
684 'compact', 'explicit' or None/'nostyle' (default), or a callable. `idstyle` has no effect if no `fixture_ref`
685 are present in the argvalues. As opposed to `ids`, a callable provided here will receive a `ParamAlternative`
686 object indicating which generated fixture should be used. See `ParamIdMakers`.
687 :param scope: The scope of the union fixture to create if `fixture_ref`s are found in the argvalues. Otherwise same
688 as in pytest.mark.parametrize.
689 :param hook: an optional hook to apply to each fixture function that is created during this call. The hook function
690 will be called every time a fixture is about to be created. It will receive a single argument (the function
691 implementing the fixture) and should return the function to use. For example you can use `saved_fixture` from
692 `pytest-harvest` as a hook in order to save all such created fixtures in the fixture store.
693 :param debug: print debug messages on stdout to analyze fixture creation (use pytest -s to see them)
694 :param args: additional {argnames: argvalues} definition
695 :return:
696 """
697 _decorate, needs_inject = _parametrize_plus(argnames, argvalues, indirect=indirect, ids=ids, idgen=idgen,
698 auto_refs=auto_refs, idstyle=idstyle, scope=scope,
699 hook=hook, debug=debug, **args)
700 if needs_inject:
701 @inject_host
702 def _apply_parametrize_plus(f, host_class_or_module):
703 return _decorate(f, host_class_or_module)
704 return _apply_parametrize_plus
705 else:
706 return _decorate
