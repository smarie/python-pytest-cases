37 # A decorator that will work to create a fixture containing 'yield', whatever the pytest version, and supports hooks
38 if PYTEST3_OR_GREATER:
39 def pytest_fixture(hook=None, **kwargs):
40 def _decorate(f):
41 # call hook if needed
42 if hook is not None:
43 f = hook(f)
44  
45 # create the fixture
46 return pytest.fixture(**kwargs)(f)
47 return _decorate

49 def pytest_fixture(hook=None, name=None, **kwargs):
50 """Generator-aware pytest.fixture decorator for legacy pytest versions"""
51 def _decorate(f):
52 if name is not None:
53 # 'name' argument is not supported in this old version, use the __name__ trick.
54 f.__name__ = name
55  
56 # call hook if needed
57 if hook is not None:
58 f = hook(f)
59  
60 # create the fixture
61 if isgeneratorfunction(f):
62 return pytest.yield_fixture(**kwargs)(f)
63 else:
64 return pytest.fixture(**kwargs)(f)
65 return _decorate

103 def list_all_fixtures_in(cls_or_module, return_names=True, recurse_to_module=False):
104 """
105 Returns a list containing all fixture names (or symbols if `return_names=False`)
106 in the given class or module.
107  
108 Note that `recurse_to_module` can be used so that the fixtures in the parent
109 module of a class are listed too.
110  
111 :param cls_or_module:
112 :param return_names:
113 :param recurse_to_module:
114 :return:
115 """
116 res = [get_fixture_name(symb) if return_names else symb
117 for n, symb in inspect.getmembers(cls_or_module, lambda f: inspect.isfunction(f) or inspect.ismethod(f))
118 if is_fixture(symb)]
119  
120 if recurse_to_module and not inspect.ismodule(cls_or_module):
121 # TODO currently this only works for a single level of nesting, we should use __qualname__ (py3) or .im_class
122 host = import_module(cls_or_module.__module__)
123 res += list_all_fixtures_in(host, recurse_to_module=True, return_names=return_names)
124  
125 return res

262 def make_test_ids(global_ids, id_marks, argnames=None, argvalues=None, precomputed_ids=None):
263 """
264 Creates the proper id for each test based on (higher precedence first)
265  
266 - any specific id mark from a `pytest.param` (`id_marks`)
267 - the global `ids` argument of pytest parametrize (`global_ids`)
268 - the name and value of parameters (`argnames`, `argvalues`) or the precomputed ids(`precomputed_ids`)
269  
270 See also _pytest.python._idvalset method
271  
272 :param global_ids:
273 :param id_marks:
274 :param argnames:
275 :param argvalues:
276 :param precomputed_ids:
277 :return:
278 """
279 if global_ids is not None:
280 # overridden at global pytest.mark.parametrize level - this takes precedence.
281 # resolve possibly infinite generators of ids here
282 p_ids = resolve_ids(global_ids, argvalues, full_resolve=True)
283 else:
284 # default: values-based
285 if precomputed_ids is not None:
286 if argnames is not None or argvalues is not None:
287 raise ValueError("Only one of `precomputed_ids` or argnames/argvalues should be provided.")
288 p_ids = precomputed_ids
289 else:
290 p_ids = make_test_ids_from_param_values(argnames, argvalues)
291  
292 # Finally, local pytest.param takes precedence over everything else
293 for i, _id in enumerate(id_marks):
294 if _id is not None:
295 p_ids[i] = _id
296 return p_ids

299 def resolve_ids(ids, # type: Optional[Union[Callable, Iterable[str]]]
300 argvalues, # type: Sized(Any)
301 full_resolve=False # type: bool
302 ):
303 # type: (...) -> Union[List[str], Callable]
304 """
305 Resolves the `ids` argument of a parametrized fixture.
306  
307 If `full_resolve` is False (default), iterable ids will be resolved, but not callable ids. This is useful if the
308 `argvalues` have not yet been cleaned of possible `pytest.param` wrappers.
309  
310 If `full_resolve` is True, callable ids will be called using the argvalues, so the result is guaranteed to be a
311 list.
312 """
313 try:
314 # an explicit list or generator of ids ?
315 iter(ids)
316 except TypeError:
317 # a callable to apply on the values
318 if full_resolve:
319 return [ids(v) for v in argvalues]
320 else:
321 # return the callable without resolving
322 return ids
323 else:
324 # iterable.
325 try:
326 # a sized container ? (list, set, tuple)
327 nb_ids = len(ids)
328 # convert to list
329 ids = list(ids)
330 except TypeError:
331 # a generator. Consume it
332 ids = [id for id, v in zip(ids, argvalues)]
333 nb_ids = len(ids)
334  
335 if nb_ids != len(argvalues):
336 raise ValueError("Explicit list or generator of `ids` provided has a different length (%s) than the number "
337 "of argvalues (%s). Ids provided: %r" % (len(ids), len(argvalues), ids))
338 return ids

341 def make_test_ids_from_param_values(param_names,
342 param_values,
343 ):
344 """
345 Replicates pytest behaviour to generate the ids when there are several parameters in a single `parametrize.
346 Note that param_values should not contain marks.
347  
348 :param param_names:
349 :param param_values:
350 :return: a list of param ids
351 """
352 if isinstance(param_names, string_types):
353 raise TypeError("param_names must be an iterable. Found %r" % param_names)
354  
355 nb_params = len(param_names)
356 if nb_params == 0:
357 raise ValueError("empty list provided")
358 elif nb_params == 1:
359 paramids = []
360 for _idx, v in enumerate(param_values):
361 _id = mini_idvalset(param_names, (v,), _idx)
362 paramids.append(_id)
363 else:
364 paramids = []
365 for _idx, vv in enumerate(param_values):
366 if len(vv) != nb_params:
367 raise ValueError("Inconsistent lengths for parameter names and values: '%s' and '%s'"
368 "" % (param_names, vv))
369 _id = mini_idvalset(param_names, vv, _idx)
370 paramids.append(_id)
371 return paramids

408 def extract_parameterset_info(argnames, argvalues, check_nb=True):
409 """
410  
411 :param argnames: the names in this parameterset
412 :param argvalues: the values in this parameterset
413 :param check_nb: a bool indicating if we should raise an error if len(argnames) > 1 and any argvalue has
414 a different length than len(argnames)
415 :return:
416 """
417 pids = []
418 pmarks = []
419 pvalues = []
420 if isinstance(argnames, string_types):
421 raise TypeError("argnames must be an iterable. Found %r" % argnames)
422 nbnames = len(argnames)
423 for v in argvalues:
424 _pid, _pmark, _pvalue = extract_pset_info_single(nbnames, v)
425  
426 pids.append(_pid)
427 pmarks.append(_pmark)
428 pvalues.append(_pvalue)
429  
430 if check_nb and nbnames > 1 and (len(_pvalue) != nbnames):
431 raise ValueError("Inconsistent number of values in pytest parametrize: %s items found while the "
432 "number of parameters is %s: %s." % (len(_pvalue), nbnames, _pvalue))
433  
434 return pids, pmarks, pvalues

437 def extract_pset_info_single(nbnames, argvalue):
438 """Return id, marks, value"""
439 if is_marked_parameter_value(argvalue):
440 # --id
441 _id = get_marked_parameter_id(argvalue)
442 # --marks
443 marks = get_marked_parameter_marks(argvalue)
444 # --value(a tuple if this is a tuple parameter)
445 argvalue = get_marked_parameter_values(argvalue, nbargs=nbnames)
446 return _id, marks, argvalue[0] if nbnames == 1 else argvalue
447 else:
448 # normal argvalue
449 return None, None, argvalue

631 def getfuncargnames(function, cls=None):
632 """Returns the names of a function's mandatory arguments."""
633 parameters = signature(function).parameters
634  
635 arg_names = tuple(
636 p.name
637 for p in parameters.values()
638 if (
639 p.kind is Parameter.POSITIONAL_OR_KEYWORD
640 or p.kind is Parameter.KEYWORD_ONLY
641 )
642 and p.default is Parameter.empty
643 )
644  
645 # If this function should be treated as a bound method even though
646 # it's passed as an unbound method or function, remove the first
647 # parameter name.
648 if cls and not isinstance(cls.__dict__.get(function.__name__, None), staticmethod):
649 arg_names = arg_names[1:]
650 # Remove any names that will be replaced with mocks.
651 if hasattr(function, "__wrapped__"):
652 arg_names = arg_names[num_mock_patch_args(function):]
653 return arg_names

663 class MiniMetafunc(Metafunc):
664 # noinspection PyMissingConstructor
665 def __init__(self, func):
666 from .plugin import PYTEST_CONFIG # late import to ensure config has been loaded by now
667  
668 self.config = PYTEST_CONFIG
669  
670 # self.config can be `None` if the same module is reloaded by another thread/process inside a test (parallelism)
671 # In that case, a priori we are outside the pytest main runner so we can silently ignore, this
672 # MetaFunc will not be used/read by anyone.
673 # See https://github.com/smarie/python-pytest-cases/issues/242
674 #
675 # if self.config is None:
676 # if pytest_is_running():
677 # raise ValueError("Internal error - config has not been correctly loaded. Please report")
678  
679 self.function = func
680 self.definition = MiniFuncDef(func.__name__)
681 self._calls = []
682 # non-default parameters
683 self.fixturenames = getfuncargnames(func)
684 # add declared used fixtures with @pytest.mark.usefixtures
685 self.fixturenames_not_in_sig = [f for f in get_pytest_usefixture_marks(func) if f not in self.fixturenames]
686 if self.fixturenames_not_in_sig:
687 self.fixturenames = tuple(self.fixturenames_not_in_sig + list(self.fixturenames))
688 # get parametrization marks
689 self.pmarks = get_pytest_parametrize_marks(self.function)
690 if self.is_parametrized:
691 self.update_callspecs()
692 # preserve order
693 self.required_fixtures = tuple(f for f in self.fixturenames if f not in self._calls[0].funcargs)
694 else:
695 self.required_fixtures = self.fixturenames

697 @property
698 def is_parametrized(self):
699 return len(self.pmarks) > 0
700  
701 @property
702 def requires_fixtures(self):
703 return len(self.required_fixtures) > 0

705 def update_callspecs(self):
706 """
707  
708 :return:

785 def cart_product_pytest(argnames, argvalues):
786 """
787 - do NOT use `itertools.product` as it fails to handle MarkDecorators
788 - we also unpack tuples associated with several argnames ("a,b") if needed
789 - we also propagate marks
790  
791 :param argnames:
792 :param argvalues:
793 :return:
794 """
795 # transform argnames into a list of lists
796 argnames_lists = [get_param_argnames_as_list(_argnames) if len(_argnames) > 0 else [] for _argnames in argnames]
797  
798 # make the cartesian product per se
799 argvalues_prod = _cart_product_pytest(argnames_lists, argvalues)
800  
801 # flatten the list of argnames
802 argnames_list = [n for nlist in argnames_lists for n in nlist]
803  
804 # apply all marks to the arvalues
805 argvalues_prod = [make_marked_parameter_value(tuple(argvalues), marks=marks) if len(marks) > 0 else tuple(argvalues)
806 for marks, argvalues in argvalues_prod]
807  
808 return argnames_list, argvalues_prod

811 def _cart_product_pytest(argnames_lists, argvalues):
812 result = []
813  
814 # first perform the sub cartesian product with entries [1:]
815 sub_product = _cart_product_pytest(argnames_lists[1:], argvalues[1:]) if len(argvalues) > 1 else None
816  
817 # then do the final product with entry [0]
818 for x in argvalues[0]:
819 # handle x
820 nb_names = len(argnames_lists[0])
821  
822 # (1) extract meta-info
823 x_id, x_marks, x_value = extract_pset_info_single(nb_names, x)
824 x_marks_lst = list(x_marks) if x_marks is not None else []
825 if x_id is not None:
826 raise ValueError("It is not possible to specify a sub-param id when using the new parametrization style. "
827 "Either use the traditional style or customize all ids at once in `idgen`")
828  
829 # (2) possibly unpack
830 if nb_names > 1:
831 # if lazy value, we have to do something
832 if is_lazy_value(x_value):
833 x_value_lst = x_value.as_lazy_items_list(nb_names)
834 else:
835 x_value_lst = list(x_value)
836 else:
837 x_value_lst = [x_value]
838  
839 # product
840 if len(argvalues) > 1:
841 for m, p in sub_product:
842 # combine marks and values
843 result.append((x_marks_lst + m, x_value_lst + p))
844 else:
845 result.append((x_marks_lst, x_value_lst))
846
