72 class _ParametrizationMark:
73 """
74 Container for the mark information that we grab from the fixtures (`@fixture`)
75  
76 Represents the information required by `@fixture` to work.
77 """
78 __slots__ = "param_names", "param_values", "param_ids"
79  
80 def __init__(self, mark):
81 bound = get_parametrize_signature().bind(*mark.args, **mark.kwargs)
82 try:
83 remaining_kwargs = bound.arguments['kwargs']
84 except KeyError:
85 pass
86 else:
87 if len(remaining_kwargs) > 0:
88 warnings.warn("parametrize kwargs not taken into account: %s. Please report it at"
89 " https://github.com/smarie/python-pytest-cases/issues" % remaining_kwargs)
90 self.param_names = get_param_argnames_as_list(bound.arguments['argnames'])
91 self.param_values = bound.arguments['argvalues']
92 try:
93 bound.apply_defaults()
94 self.param_ids = bound.arguments['ids']
95 except AttributeError:
96 # can happen if signature is from funcsigs so we have to apply ourselves
97 self.param_ids = bound.arguments.get('ids', None)

110 def copy_pytest_marks(from_f, to_f, override=False):
111 """Copy all pytest marks from a function or class to another"""
112 from_marks = get_pytest_marks_on_function(from_f)
113 to_marks = [] if override else get_pytest_marks_on_function(to_f)
114 # note: the new marks are appended *after* existing if no override
115 to_f.pytestmark = to_marks + from_marks

132 def get_pytest_marks_on_function(f,
133 as_decorators=False # type: bool
134 ):
135 # type: (...) -> Union[List[Mark], List[MarkDecorator]]
136 """
137 Utility to return a list of *ALL* pytest marks (not only parametrization) applied on a function
138 Note that this also works on classes
139  
140 :param f:
141 :param as_decorators: transforms the marks into decorators before returning them
142 :return:
143 """
144 try:
145 mks = f.pytestmark
146 except AttributeError:
147 try:
148 # old pytest < 3: marks are set as fields on the function object
149 # but they do not have a particulat type, their type is 'instance'...
150 mks = [v for v in vars(f).values() if str(v).startswith("<MarkInfo '")]
151 except AttributeError:
152 return []
153  
154 # in the new version of pytest the marks have to be transformed into decorators explicitly
155 if as_decorators:
156 return markinfos_to_markdecorators(mks, function_marks=True)
157 else:
158 return mks

162 """lists all marks on an item such as `request._pyfuncitem`"""

169 def get_pytest_usefixture_marks(f):
170 # pytest > 3.2.0
171 marks = getattr(f, 'pytestmark', None)
172 if marks is not None:
173 return tuple(itertools.chain.from_iterable(
174 mark.args for mark in marks if mark.name == 'usefixtures'
175 ))
176 else:
177 # older versions
178 mark_info = getattr(f, 'usefixtures', None)
179 if mark_info is not None:
180 return mark_info.args
181 else:
182 return ()

186 marks = getattr(f, 'pytestmark', None)
187 if marks is not None:
188 # pytest > 3.2.0
189 new_marks = [m for m in marks if m.name != mark_name]
  • B010 Do not call setattr with a constant attribute value, it is not any safer than normal property access.

192 # older versions

200 def get_pytest_parametrize_marks(f):
201 """
202 Returns the @pytest.mark.parametrize marks associated with a function (and only those)
203  
204 :param f:
205 :return: a tuple containing all 'parametrize' marks
206 """
207 # pytest > 3.2.0
208 marks = getattr(f, 'pytestmark', None)
209 if marks is not None:
210 return tuple(_ParametrizationMark(m) for m in marks if m.name == 'parametrize')
211 else:
212 # older versions
213 mark_info = getattr(f, 'parametrize', None)
214 if mark_info is not None:
215 # mark_info.args contains a list of (name, values)
216 if len(mark_info.args) % 2 != 0:
217 raise ValueError("internal pytest compatibility error - please report")
218 nb_parametrize_decorations = len(mark_info.args) // 2
219 if nb_parametrize_decorations > 1 and len(mark_info.kwargs) > 0:
220 raise ValueError("Unfortunately with this old pytest version it is not possible to have several "
221 "parametrization decorators while specifying **kwargs, as all **kwargs are "
222 "merged, leading to inconsistent results. Either upgrade pytest, remove the **kwargs,"
223 "or merge all the @parametrize decorators into a single one. **kwargs: %s"
224 % mark_info.kwargs)
225 res = []
226 for i in range(nb_parametrize_decorations):
227 param_name, param_values = mark_info.args[2*i:2*(i+1)]
228 res.append(_ParametrizationMark(_LegacyMark(param_name, param_values, **mark_info.kwargs)))
229 return tuple(res)
230 else:
231 return ()

234 # ---- tools to reapply marks on test parameter values, whatever the pytest version ----
235  
236 # Compatibility for the way we put marks on single parameters in the list passed to @pytest.mark.parametrize
237 # see https://docs.pytest.org/en/3.3.0/skipping.html?highlight=mark%20parametrize#skip-xfail-with-parametrize
238  
239 # check if pytest.param exists
240 has_pytest_param = hasattr(pytest, 'param')
241  
242  
243 if not has_pytest_param:
244 # if not this is how it was done
245 # see e.g. https://docs.pytest.org/en/2.9.2/skipping.html?highlight=mark%20parameter#skip-xfail-with-parametrize
246 def make_marked_parameter_value(argvalues_tuple, marks):
247 if len(marks) > 1:
248 raise ValueError("Multiple marks on parameters not supported for old versions of pytest")
249 else:
250 if not isinstance(argvalues_tuple, tuple):
251 raise TypeError("argvalues must be a tuple !")
252  
253 # get a decorator for each of the markinfo
254 marks_mod = markinfos_to_markdecorators(marks, function_marks=False)
255  
256 # decorate. We need to distinguish between single value and multiple values
257 # indeed in pytest 2 a single arg passed to the decorator is passed directly
258 # (for example: @pytest.mark.skip(1) in parametrize)
259 return marks_mod[0](argvalues_tuple) if len(argvalues_tuple) > 1 else marks_mod[0](argvalues_tuple[0])
260 else:
261 # Otherwise pytest.param exists, it is easier
262 def make_marked_parameter_value(argvalues_tuple, marks):
263 if not isinstance(argvalues_tuple, tuple):
264 raise TypeError("argvalues must be a tuple !")
265  
266 # get a decorator for each of the markinfo
267 marks_mod = markinfos_to_markdecorators(marks, function_marks=False)
268  
269 # decorate
270 return pytest.param(*argvalues_tuple, marks=marks_mod)

273 def markinfos_to_markdecorators(marks, # type: Iterable[Mark]
274 function_marks=False # type: bool
275 ):
276 # type: (...) -> List[MarkDecorator]
277 """
278 Transforms the provided marks (MarkInfo or Mark in recent pytest) obtained from marked cases, into MarkDecorator so
279 that they can be re-applied to generated pytest parameters in the global @pytest.mark.parametrize.
280  
281 Returns a list.
282  
283 :param marks:
284 :param function_marks:
285 :return:
286 """
287 marks_mod = []
288 try:
289 # suppress the warning message that pytest generates when calling pytest.mark.MarkDecorator() directly
290 with warnings.catch_warnings():
291 warnings.simplefilter("ignore")
292 for m in marks:
293 # create a dummy new MarkDecorator named "MarkDecorator" for reference
294 md = pytest.mark.MarkDecorator()
295  
296 if PYTEST3_OR_GREATER:
297 if isinstance(m, type(md)):
298 # already a decorator, we can use it
299 marks_mod.append(m)
300 else:
301 md.mark = m
302 marks_mod.append(md)
303 else:
304 # always recreate one, type comparison does not work (all generic stuff)
305 md.name = m.name
306 # md.markname = m.name
307 if function_marks:
308 md.args = m.args # a mark on a function does not include the function in the args
309 else:
310 md.args = m.args[:-1] # not a function: the value is in the args, remove it
311 md.kwargs = m.kwargs
312  
313 # markinfodecorator = getattr(pytest.mark, markinfo.name)
314 # markinfodecorator(*markinfo.args)
315  
316 marks_mod.append(md)
317  
318 except Exception as e:
319 warnings.warn("Caught exception while trying to mark case: [%s] %s" % (type(e), e))
320 return marks_mod
