35 def param_fixture(argname, # type: str
36 argvalues, # type: Iterable[Any]
37 autouse=False, # type: bool
38 ids=None, # type: Union[Callable, Iterable[str]]
39 scope="function", # type: str
40 hook=None, # type: Callable[[Callable], Callable]
41 debug=False, # type: bool
42 **kwargs):
43 """
44 Identical to `param_fixtures` but for a single parameter name, so that you can assign its output to a single
45 variable.
46  
47 ```python
48 import pytest
49 from pytest_cases import param_fixtures, param_fixture
50  
51 # create a single parameter fixture
52 my_parameter = param_fixture("my_parameter", [1, 2, 3, 4])
53  
54 @pytest.fixture
55 def fixture_uses_param(my_parameter):
56 ...
57  
58 def test_uses_param(my_parameter, fixture_uses_param):
59 ...
60 ```
61  
62 :param argname: see fixture `name`
63 :param argvalues: see fixture `params`
64 :param autouse: see fixture `autouse`
65 :param ids: see fixture `ids`
66 :param scope: see fixture `scope`
67 :param hook: an optional hook to apply to each fixture function that is created during this call. The hook function
68 will be called everytime a fixture is about to be created. It will receive a single argument (the function
69 implementing the fixture) and should return the function to use. For example you can use `saved_fixture` from
70 `pytest-harvest` as a hook in order to save all such created fixtures in the fixture store.
71 :param debug: print debug messages on stdout to analyze fixture creation (use pytest -s to see them)
72 :param kwargs: any other argument for 'fixture'
73 :return: the create fixture
74 """

77 "parameter name. Use `param_fixtures` instead - but note that it creates several fixtures.")
78 elif len(argname.replace(' ', '')) == 0:
79 raise ValueError("empty argname")

81 # todo what if this is called in a class ?
82 caller_module = get_caller_module()
83  
84 return _create_param_fixture(caller_module, argname, argvalues, autouse=autouse, ids=ids, scope=scope,
85 hook=hook, debug=debug, **kwargs)

137 def param_fixtures(argnames, # type: str
138 argvalues, # type: Iterable[Any]
139 autouse=False, # type: bool
140 ids=None, # type: Union[Callable, Iterable[str]]
141 scope="function", # type: str
142 hook=None, # type: Callable[[Callable], Callable]
143 debug=False, # type: bool
144 **kwargs):
145 """
146 Creates one or several "parameters" fixtures - depending on the number or coma-separated names in `argnames`. The
147 created fixtures are automatically registered into the callers' module, but you may wish to assign them to
148 variables for convenience. In that case make sure that you use the same names, e.g.
149 `p, q = param_fixtures('p,q', [(0, 1), (2, 3)])`.
150  
151 Note that the (argnames, argvalues, ids) signature is similar to `@pytest.mark.parametrize` for consistency,
152 see https://docs.pytest.org/en/latest/reference.html?highlight=pytest.param#pytest-mark-parametrize
153  
154 ```python
155 import pytest
156 from pytest_cases import param_fixtures, param_fixture
157  
158 # create a 2-tuple parameter fixture
159 arg1, arg2 = param_fixtures("arg1, arg2", [(1, 2), (3, 4)])
160  
161 @pytest.fixture
162 def fixture_uses_param2(arg2):
163 ...
164  
165 def test_uses_param2(arg1, arg2, fixture_uses_param2):
166 ...
167 ```
168  
169 :param argnames: same as `@pytest.mark.parametrize` `argnames`.
170 :param argvalues: same as `@pytest.mark.parametrize` `argvalues`.
171 :param autouse: see fixture `autouse`
172 :param ids: same as `@pytest.mark.parametrize` `ids`
173 :param scope: see fixture `scope`
174 :param hook: an optional hook to apply to each fixture function that is created during this call. The hook function
175 will be called everytime a fixture is about to be created. It will receive a single argument (the function
176 implementing the fixture) and should return the function to use. For example you can use `saved_fixture` from
177 `pytest-harvest` as a hook in order to save all such created fixtures in the fixture store.
178 :param debug: print debug messages on stdout to analyze fixture creation (use pytest -s to see them)
179 :param kwargs: any other argument for the created 'fixtures'
180 :return: the created fixtures
181 """
182 argnames_lst = get_param_argnames_as_list(argnames)
183  
184 caller_module = get_caller_module()
185  
186 if len(argnames_lst) < 2:
187 return _create_param_fixture(caller_module, argnames, argvalues, autouse=autouse, ids=ids, scope=scope,
188 hook=hook, debug=debug, **kwargs)
189 else:
190 return _create_params_fixture(caller_module, argnames_lst, argvalues, autouse=autouse, ids=ids, scope=scope,
191 hook=hook, debug=debug, **kwargs)

201 debug=False, # type: bool
202 **kwargs):
203 argnames = ','.join(argnames_lst)
204 created_fixtures = []
205  
206 # create the root fixture that will contain all parameter values
207 # note: we sort the list so that the first in alphabetical order appears first. Indeed pytest uses this order.
208 root_fixture_name = "%s__param_fixtures_root" % ('_'.join(sorted(argnames_lst)))
209  
210 # Dynamically add fixture to caller's module as explained in https://github.com/pytest-dev/pytest/issues/2424
211 root_fixture_name = check_name_available(fixtures_dest, root_fixture_name, if_name_exists=CHANGE,
212 caller=param_fixtures)
213  
214 if debug:
  • T001 Print found.
215 print("Creating parametrized 'root' fixture %r returning %r" % (root_fixture_name, argvalues))
216  
217 @fixture(name=root_fixture_name, autouse=autouse, scope=scope, hook=hook, **kwargs)
218 @pytest.mark.parametrize(argnames, argvalues, ids=ids)
219 @with_signature("%s(%s)" % (root_fixture_name, argnames))
220 def _root_fixture(**_kwargs):
221 return tuple(_kwargs[k] for k in argnames_lst)
222  
223 # Override once again the symbol with the correct contents
224 setattr(fixtures_dest, root_fixture_name, _root_fixture)
225  
226 # finally create the sub-fixtures
227 for param_idx, argname in enumerate(argnames_lst):
228 # create the fixture
229 # To fix late binding issue with `param_idx` we add an extra layer of scope: a factory function
230 # See https://stackoverflow.com/questions/3431676/creating-functions-in-a-loop
231 def _create_fixture(_param_idx):
232  
233 if debug:
  • T001 Print found.
234 print("Creating nonparametrized 'view' fixture %r returning %r[%s]"
235 % (argname, root_fixture_name, _param_idx))
236  
237 @fixture(name=argname, scope=scope, autouse=autouse, hook=hook, **kwargs)
238 @with_signature("%s(%s)" % (argname, root_fixture_name))
239 def _param_fixture(**_kwargs):
240 params = _kwargs.pop(root_fixture_name)
241 return params[_param_idx]
242  
243 return _param_fixture
244  
245 # create it
246 fix = _create_fixture(param_idx)
247  
248 # add to module
249 check_name_available(fixtures_dest, argname, if_name_exists=WARN, caller=param_fixtures)
250 setattr(fixtures_dest, argname, fix)
251  
252 # collect to return the whole list eventually
253 created_fixtures.append(fix)
254  
255 return created_fixtures

282 def fixture(scope="function", # type: str
283 autouse=False, # type: bool
284 name=None, # type: str
285 unpack_into=None, # type: Iterable[str]
286 hook=None, # type: Callable[[Callable], Callable]
287 fixture_func=DECORATED, # noqa
288 **kwargs):
289 """ decorator to mark a fixture factory function.
290  
291 Identical to `@pytest.fixture` decorator, except that
292  
293 - when used in a fixture union (either explicit `fixture_union` or indirect through `@parametrize`+`fixture_ref`
294 or `@parametrize_with_cases`), it will not be setup/teardown unnecessarily in tests that do not require it.
295  
296 - it supports multi-parametrization with `@pytest.mark.parametrize` as requested in
297 https://github.com/pytest-dev/pytest/issues/3960. As a consequence it does not support the `params` and `ids`
298 arguments anymore.
299  
300 - it supports a new argument `unpack_into` where you can provide names for fixtures where to unpack this fixture
301 into.
302  
303 As a consequence it does not support the `params` and `ids` arguments anymore.
304  
305 :param scope: the scope for which this fixture is shared, one of "function" (default), "class", "module" or
306 "session".
307 :param autouse: if True, the fixture func is activated for all tests that can see it. If False (the default) then
308 an explicit reference is needed to activate the fixture.
309 :param name: the name of the fixture. This defaults to the name of the decorated function. Note: If a fixture is
310 used in the same module in which it is defined, the function name of the fixture will be shadowed by the
311 function arg that requests the fixture; one way to resolve this is to name the decorated function
312 ``fixture_<fixturename>`` and then use ``@pytest.fixture(name='<fixturename>')``.
313 :param unpack_into: an optional iterable of names, or string containing coma-separated names, for additional
314 fixtures to create to represent parts of this fixture. See `unpack_fixture` for details.
315 :param hook: an optional hook to apply to each fixture function that is created during this call. The hook function
316 will be called everytime a fixture is about to be created. It will receive a single argument (the function
317 implementing the fixture) and should return the function to use. For example you can use `saved_fixture` from
318 `pytest-harvest` as a hook in order to save all such created fixtures in the fixture store.
319 :param kwargs: other keyword arguments for `@pytest.fixture`
320 """
321 # todo what if this is called in a class ?
322  
323 # the offset is 3 because of @function_decorator (decopatch library)
324 return _decorate_fixture_plus(fixture_func, scope=scope, autouse=autouse, name=name, unpack_into=unpack_into,
325 hook=hook, _caller_module_offset_when_unpack=3, **kwargs)

356 def _decorate_fixture_plus(fixture_func,
357 scope="function", # type: str
358 autouse=False, # type: bool
359 name=None, # type: str
360 unpack_into=None, # type: Iterable[str]
361 hook=None, # type: Callable[[Callable], Callable]
362 _caller_module_offset_when_unpack=3, # type: int
363 **kwargs):
364 """ decorator to mark a fixture factory function.
365  
366 Identical to `@pytest.fixture` decorator, except that
367  
368 - it supports multi-parametrization with `@pytest.mark.parametrize` as requested in
369 https://github.com/pytest-dev/pytest/issues/3960. As a consequence it does not support the `params` and `ids`
370 arguments anymore.
371  
372 - it supports a new argument `unpack_into` where you can provide names for fixtures where to unpack this fixture
373 into.
374  
375 :param scope: the scope for which this fixture is shared, one of "function" (default), "class", "module" or
376 "session".
377 :param autouse: if True, the fixture func is activated for all tests that can see it. If False (the default) then
378 an explicit reference is needed to activate the fixture.
379 :param name: the name of the fixture. This defaults to the name of the decorated function. Note: If a fixture is
380 used in the same module in which it is defined, the function name of the fixture will be shadowed by the
381 function arg that requests the fixture; one way to resolve this is to name the decorated function
382 ``fixture_<fixturename>`` and then use ``@pytest.fixture(name='<fixturename>')``.
383 :param unpack_into: an optional iterable of names, or string containing coma-separated names, for additional
384 fixtures to create to represent parts of this fixture. See `unpack_fixture` for details.
385 :param hook: an optional hook to apply to each fixture function that is created during this call. The hook function
386 will be called everytime a fixture is about to be created. It will receive a single argument (the function
387 implementing the fixture) and should return the function to use. For example you can use `saved_fixture` from
388 `pytest-harvest` as a hook in order to save all such created fixtures in the fixture store.
389 :param kwargs: other keyword arguments for `@pytest.fixture`
390 """
391 if name is not None:
392 # Compatibility for the 'name' argument
393 if PYTEST3_OR_GREATER:
394 # pytest version supports "name" keyword argument
395 kwargs['name'] = name
396 elif name is not None:
397 # 'name' argument is not supported in this old version, use the __name__ trick.
398 fixture_func.__name__ = name
399  
400 # if unpacking is requested, do it first
401 if unpack_into is not None:
402 # get the future fixture name if needed
403 if name is None:
404 name = fixture_func.__name__
405  
406 # get caller module to create the symbols
407 caller_module = get_caller_module(frame_offset=_caller_module_offset_when_unpack)
408  
409 # note that we cannot use in_cls=True since we have no way to assign the unpacked fixtures to the class
410 _make_unpack_fixture(caller_module, unpack_into, name, hook=hook, in_cls=False)
411  
412 # (1) Collect all @pytest.mark.parametrize markers (including those created by usage of @cases_data)
413 parametrizer_marks = get_pytest_parametrize_marks(fixture_func)
414 if len(parametrizer_marks) < 1:
415 # make the fixture union-aware
416 wrapped_fixture_func = ignore_unused(fixture_func)
417  
418 # resolve possibly infinite generators of ids here
419 if 'params' in kwargs and 'ids' in kwargs:
420 kwargs['ids'] = resolve_ids(kwargs['ids'], kwargs['params'], full_resolve=False)
421  
422 # transform the created wrapper into a fixture
423 return pytest_fixture(scope=scope, autouse=autouse, hook=hook, **kwargs)(wrapped_fixture_func)
424  
425 else:
426 if 'params' in kwargs:
427 raise ValueError(
428 "With `fixture` you cannot mix usage of the keyword argument `params` and of "
429 "the pytest.mark.parametrize marks")
430  
431 # (2) create the huge "param" containing all params combined
432 # --loop (use the same order to get it right)
433 param_defs = []
434 params_values = []
435 params_ids = []
436 params_marks = []
437 for pmark in parametrizer_marks:
438 # -- pmark is a single @pytest.parametrize mark. --
439  
440 # check number of parameter names in this parameterset
441 if len(pmark.param_names) < 1:
442 raise ValueError("Fixture function '%s' decorated with '@fixture' has an empty parameter "
443 "name in a @pytest.mark.parametrize mark")
444  
445 # remember the argnames
446 param_defs.append(FixtureParam(pmark.param_names))
447  
448 # separate specific configuration (pytest.param()) from the values
449 custom_pids, _pmarks, _pvalues = extract_parameterset_info(pmark.param_names, pmark.param_values, check_nb=True)
450  
451 # get the ids by merging/creating the various possibilities
452 _paramids = make_test_ids(argnames=pmark.param_names, argvalues=_pvalues,
453 global_ids=pmark.param_ids, id_marks=custom_pids)
454  
455 # Finally store the ids, marks, and values for this parameterset
456 params_ids.append(_paramids)
457 params_marks.append(tuple(_pmarks))
458 params_values.append(tuple(_pvalues))
459  
460 # (3) generate the ids and values, possibly reapplying marks
461 if len(param_defs) == 1:
462 # A single @parametrize : we can simplify - that will be more readable
463 final_ids = params_ids[0]
464 final_marks = params_marks[0]
465 # note: we dot his even for a single @parametrize as it allows `current_case` to get the parameter names easily
466 final_values = [CombinedFixtureParamValue(param_defs, (v,)) for v in params_values[0]]
467  
468 # reapply the marks
469 for i, marks in enumerate(final_marks):
470 if marks is not None:
471 final_values[i] = make_marked_parameter_value((final_values[i],), marks=marks)
472 else:
473 # Multiple @parametrize: since pytest does not support several, we merge them with "apparence" of several
474 # --equivalent id
475 final_ids = combine_ids(product(*params_ids))
476 # --merge all values, we'll unpack them in the wrapper below
477 final_values = [CombinedFixtureParamValue(param_defs, v) for v in product(*params_values)]
478  
479 final_marks = tuple(product(*params_marks))
480  
481 # reapply the marks
482 for i, marks in enumerate(final_marks):
483 ms = [m for mm in marks if mm is not None for m in mm]
484 if len(ms) > 0:
485 final_values[i] = make_marked_parameter_value((final_values[i],), marks=ms)
486  
487 if len(final_values) != len(final_ids):
488 raise ValueError("Internal error related to fixture parametrization- please report")
489  
490 # (4) wrap the fixture function so as to remove the parameter names and add 'request' if needed
491 all_param_names = tuple(v for pnames in param_defs for v in pnames.argnames)
492  
493 # --create the new signature that we want to expose to pytest
494 old_sig = signature(fixture_func)
495 for p in all_param_names:
496 if p not in old_sig.parameters:
497 raise ValueError("parameter '%s' not found in fixture signature '%s%s'"
498 "" % (p, fixture_func.__name__, old_sig))
499 new_sig = remove_signature_parameters(old_sig, *all_param_names)
500 # add request if needed
501 func_needs_request = 'request' in old_sig.parameters
502 if not func_needs_request:
503 # Add it last so that `self` argument in class functions remains the first
504 new_sig = add_signature_parameters(new_sig, last=Parameter('request', kind=Parameter.POSITIONAL_OR_KEYWORD))
505  
506 # --common routine used below. Fills kwargs with the appropriate names and values from fixture_params
507 def _map_arguments(*_args, **_kwargs):
508 request = _kwargs['request'] if func_needs_request else _kwargs.pop('request')
509  
510 # sanity check: we have created this combined value in the combined parametrization.
511 _paramz = request.param
512 if not isinstance(_paramz, CombinedFixtureParamValue):
513 # This can happen when indirect parametrization has been used.
514 # In that case we can work but this parameter will not appear in `current_cases` fixture
515 _paramz = CombinedFixtureParamValue(param_defs, _paramz if len(param_defs) > 1 else (_paramz,))
516  
517 # populate the parameters
518 for p_names, p_argvals in _paramz.iterparams():
519 if len(p_names) == 1:
520 # a single parameter for that generated fixture (@pytest.mark.parametrize with a single name)
521 _kwargs[p_names[0]] = get_lazy_args(p_argvals, request)
522 else:
523 # several parameters for that generated fixture (@pytest.mark.parametrize with several names)
524 # unpack all of them and inject them in the kwargs
525 for old_p_name, old_p_value in zip(p_names, p_argvals):
526 _kwargs[old_p_name] = get_lazy_args(old_p_value, request)
527  
528 return _args, _kwargs
529  
530 # --Finally create the fixture function, a wrapper of user-provided fixture with the new signature
531 if not isgeneratorfunction(fixture_func):
532 # normal function with return statement
533 @wraps(fixture_func, new_sig=new_sig)
534 def wrapped_fixture_func(*_args, **_kwargs):
535 if not is_used_request(_kwargs['request']):
